96)query 1:

SELECT 
    s1.id,
    COALESCE(s2.student, s1.student) AS student
FROM
    Seat s1
LEFT JOIN
    Seat s2
    ON (s1.id % 2 = 1 AND s2.id = s1.id + 1) 
    OR (s1.id % 2 = 0 AND s2.id = s1.id - 1)
ORDER BY 
    s1.id;

+++++++

query 2:

SELECT 
    CASE 
        WHEN id % 2 = 1 AND id + 1 <= (SELECT MAX(id) FROM Seat) THEN id + 1
        WHEN id % 2 = 0 THEN id - 1
        ELSE id
    END AS id,
    student
FROM Seat
ORDER BY id;
